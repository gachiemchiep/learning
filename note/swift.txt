Constance vs Variable
  
    # value can be changed over time
    var myVariable = 42
    # value can not be changed
    let myConstant = 50

    # specify the type by adding ": String"
    let myConstant: Double = 50
    var myVal: String	   = "test"

    # convert between type
    var myInt: Int = 5
    var myString: String = "5"
    String(myInt)
    Int(myString)
